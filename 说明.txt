# 方式1: 纯配置文件（推荐）
./server.exe

# 方式2: 指定配置文件
./server.exe -config my_config.json

# 方式3: 命令行参数覆盖配置文件中的值
./server.exe -addr :9443 -token mytoken

# 方式4: 通过 run.bat 启动
run.bat
